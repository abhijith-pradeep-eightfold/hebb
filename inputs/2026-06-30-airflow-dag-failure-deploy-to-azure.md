from internal.build_log import BuildLog

SCALAR_COLS = ['id', 'namespace', 't_create', 'git_revision', 'status', 'tag',
               'release_branch', 't_prod_deploy', 'prod_deploy_duration_sec']
start = '2026-06-29 00:00:00'
end   = '2026-07-01 00:00:00'

def rows(filter_by, label, limit=100):
    res = BuildLog().load(filter_by=filter_by, order_by='t_create desc',
                          limit=limit, columns=SCALAR_COLS, db='global', return_dict=True)
    res = res if isinstance(res, list) else ([res] if res else [])
    print('=== %s: %d row(s) ===' % (label, len(res)))
    for r in res:
        print('id=%s ns=%s tag=%s status=%r t_create=%s git=%s t_prod_deploy=%s' % (
            r.get('id'), r.get('namespace'), r.get('tag'), r.get('status'),
            r.get('t_create'), r.get('git_revision'), r.get('t_prod_deploy')))
    return res

mcp = rows({'namespace LIKE': '%mcp%', 't_create>=': start, 't_create<': end},
           'namespace LIKE %mcp% in window')
rows({'status LIKE': '%Deployment%', 't_create>=': start, 't_create<': end},
     'status LIKE %Deployment% in window', limit=100)

import json
for m in mcp:
    full = BuildLog().load(filter_by={'id': m['id']}, db='global')
    dj = (full.data_json or {}) if full else {}
    print('--- full mcp build_log id=%s ns=%s status=%r tag=%s ---' % (
        m['id'], m['namespace'], m['status'], m['tag']))
    compact = {k: (str(v)[:300] + '...[trunc]' if len(str(v)) > 300 else v) for k, v in dj.items()}
    print(json.dumps(compact, indent=2, default=str))
